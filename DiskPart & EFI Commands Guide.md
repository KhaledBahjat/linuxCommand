# 📀 DiskPart & EFI Commands Guide

## 🧩 الأوامر

```bash
diskpart              # تشغيل أداة إدارة الأقراص في ويندوز
list disk             # عرض كل الأقراص المتصلة بالجهاز
select disk *         # اختيار قرص معين (استبدل * برقم الديسك مثل 0)
list partition        # عرض البارتشنات داخل القرص المختار
select partition *    # اختيار بارتشن معين (استبدل * برقم البارتشن)
assign letter=z       # إعطاء البارتشن حرف Z عشان نقدر ندخله
exit                  # الخروج من أداة diskpart

z:                    # الدخول على البارتشن اللي خدنا حرف Z
dir                   # عرض الملفات والمجلدات داخل البارتشن
cd efi                # الدخول إلى مجلد EFI (مجلد الإقلاع Boot)
dir                   # عرض محتويات مجلد EFI
rd kali /s            # حذف مجلد kali بالكامل بكل الملفات اللي جواه
exit                  # الخروج من CMD
